categoricalColumns = ['gender','smoking_status']

for categoricalCol in categoricalColumns:
    stringIndexer = StringIndexer(inputCol = categoricalCol, outputCol = categoricalCol + 'Index')
   # encoder = OneHotEncoder(

              ]
